# SQL code for Lab 4
#Schema 1
```sql
  CREATE TABLE  Classes (
    coursID int,
    PRIMARY KEY (coursID)
   );

   CREATE TABLE Professors (
      ID int,
      PRIMARY KEY (ID)
     );

  CREATE TABLE Teaches (
    coursID int,
    ID int,
    coursID REFERENCES Classes(coursID),
    ID REFERENCES Professors(ID)
   );
```
